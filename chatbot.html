  <!-- Chatbot Logic -->

  <script>

    const chatbox = document.getElementById('chatbox');

    const form = document.getElementById('chat-form');

    const input = document.getElementById('user-input');


    const replies = [

      { keyword: "sad", response: "I'm really sorry you're feeling this way. Want to share more about it?" },

      { keyword: "anxious", response: "Take a deep breath with me... Inhale slowly... and exhale. You're safe here." },

      { keyword: "depressed", response: "It’s okay to feel this way sometimes. You’re not alone. I’m right here with you 💙" },

      { keyword: "lonely", response: "You’re not alone, I promise. Let’s talk through it, if you’re open to it." },

      { keyword: "help", response: "I’m here to help. Would you like to connect with a therapist or try a calming activity?" },

      { keyword: "panic", response: "Let’s try the 5-4-3-2-1 grounding exercise. Start by naming 5 things you can see around you." },

      { keyword: "angry", response: "Anger is valid. Want to talk about what made you feel this way?" },

      { keyword: "tired", response: "You’ve been carrying a lot. Rest is okay. You deserve to pause and breathe." },

      { keyword: "okay", response: "I'm glad to hear that. I’m still here to support you anytime you need it." },

    ];


    form.addEventListener('submit', function (e) {

      e.preventDefault();

      const userMessage = input.value.trim();

      if (!userMessage) return;


      addMessage('You', userMessage, 'right');

      input.value = '';


      setTimeout(() => {

        const botMessage = getBotReply(userMessage);

        addMessage('Tapasyaa', botMessage, 'left');

      }, 600);

    });


    function addMessage(sender, text, side) {

      const messageDiv = document.createElement('div');

      messageDiv.className = `flex ${side === 'right' ? 'justify-end' : 'justify-start'}`;

      messageDiv.innerHTML = `

        <div class="bg-${side === 'right' ? '[#A8DADC]' : '[#E0FBFC]'} max-w-[75%] p-3 rounded-lg text-[#1D3557]">

          <span class="font-semibold">${sender}:</span> ${text}

        </div>

      `;

      chatbox.appendChild(messageDiv);

      chatbox.scrollTop = chatbox.scrollHeight;

    }


    function getBotReply(input) {

      const lowerInput = input.toLowerCase();

      for (let pair of replies) {

        if (lowerInput.includes(pair.keyword)) return pair.response;

      }

      return "Thank you for sharing. I'm listening — tell me more about how you're feeling 💜";

    }

  </script>
